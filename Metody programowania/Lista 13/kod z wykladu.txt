Inductive tree (A : Type) : Type :=
| Leaf
| Node (l : tree A) (x : A) (r : tree A)
.

Arguments Leaf {A}.
Arguments Node {A}.

Fixpoint tree_map {A B : Type}
    (f : A → B) (t : tree A) : tree B :=
  match t with
  | Leaf       => Leaf
  | Node l x r => Node (tree_map f l) (f x) (tree_map f r)
  end.

Fixpoint flatten {A : Type} (t : tree A) : list A :=
  match t with
  | Leaf       => []
  | Node l x r => flatten l ++ [x] ++ flatten r
  end.

Lemma flatten_map {A B : Type} (f : A → B) t :
  flatten (tree_map f t) = map f (flatten t).
Proof.
  induction t.
  + reflexivity.
  + simpl.
    rewrite IHt1, IHt2.
    rewrite map_app.
    reflexivity.
Qed.

Fixpoint insert (x : nat) (t : tree nat) : tree nat :=
  match t with
  | Leaf       => Node Leaf x Leaf
  | Node l y r =>
    if lt_dec x y
    then Node (insert x l) y r
    else if lt_dec y x
    then Node l y (insert x r)
    else t
  end.

Fixpoint in_tree {A : Type} (x : A) (t : tree A) : Prop :=
  match t with
  | Leaf       => False
  | Node l y r => in_tree x l ∨ x = y ∨ in_tree x r
  end.

Fixpoint bst (t : tree nat) : Prop :=
  match t with
  | Leaf       => True
  | Node l x r =>
    bst l ∧ bst r ∧
    (∀ y, in_tree y l → y < x) ∧
    (∀ y, in_tree y r → x < y)
  end.

Theorem insert_spec1 x y t :
  in_tree y (insert x t) ↔ in_tree y t ∨ y = x.
Proof.
  induction t as [ | ? ? z ]; simpl.
  + tauto.
  + destruct (lt_dec x z); [ | destruct (lt_dec z x) ].
    - simpl; tauto.
    - simpl; tauto.
    - split; [ tauto | ].
      simpl; simpl.
      assert (x = z) by
        (apply Nat.le_antisymm; apply not_lt; assumption).
      subst; tauto.
Qed.

Theorem insert_bst x t :
  bst t → bst (insert x t).
Proof.
  induction t as [ | ? ? y ].
  + simpl; tauto.
  + simpl.
    intros H.
    destruct H as [ Hbst1 [ Hbst2 [ Ht1 Ht2 ] ] ].
    destruct (lt_dec x y); [ | destruct (lt_dec y x) ].
    - simpl; repeat split; auto; [].
      intros z H; apply insert_spec1 in H.
      destruct H; [ auto | ].
      subst; assumption.
    - simpl; repeat split; auto; [].
      intros z H; apply insert_spec1 in H.
      destruct H; [ auto | ].
      subst; assumption.
    - simpl; auto.
Qed.
